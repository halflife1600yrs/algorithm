### 快读

```c++
namespace io {
    const int SIZE = 1e7 + 10;
    char inbuff[SIZE];
    char *l, *r;
    inline void init() {
        l = inbuff;
        r = inbuff + fread(inbuff, 1, SIZE, stdin);
    }
    inline char gc() {
        if (l == r) init();
        return (l != r) ? *(l++) : EOF;
    }
    void read(int &x) {
        x = 0; char ch = gc();
        while(!isdigit(ch)) ch = gc();
        while(isdigit(ch)) x = x * 10 + ch - '0', ch = gc();
    }
} using io::read;
```

### 线性筛逆元

```c++
ll _inv[N];  
void pre()
{
    _inv[0]=_inv[1]=1;
    for(int i=2;i<N;i++)
        _inv[i]=((MODE-MODE/i)*_inv[MODE%i])%MODE;
}
```

### 手动实现简单queue

```c++
namespace q
{
    const int top=MAXV*2;
    int data[top],s,t;
    inline void push(int x) { data[t==top-1? t=0,top-1:t++]=x; }
    inline int pop() { return data[s==top-1? s=0,top-1:s++]; }
}
```
