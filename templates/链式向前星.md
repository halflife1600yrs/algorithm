### 链式向前星

后面的其他基于向前星的算法结构体都会继承这个类

```c++
typedef long long ll;
const int MAXV = 1e5 + 5, MAXE = 5e5 + 5; // 建模之后一定不要忘记计算新模型的边数
const ll INF = 0x7f7f7f7f7f7f7f7f;
// INF的值根据需要增大
int v, e;

struct Graph
{
    struct Edge
    {
        int to, last;
        ll l;
        void set(int _to, int _last, ll _l) { to = _to, last = _last, l = _l; }
    } edges[MAXE]; // 边记得开两倍!!!

    int top, head[MAXV];

    void init() { top = 0, memset(head, -1, sizeof(head)); }

    void add(int fr, int to, ll l)
    {
        edges[top].set(to, head[fr], l);
        head[fr] = top++;
    }
};
```

### 向前星建反图

```c++
head2[MAXV];
void rev()
{
    for(int i=1;i<=v;++i) head2[i]=-1;
    for(int i=1,to,cur,last;i<=v;++i)
    {
        last=head[i],cur=last;
        while(~last)
        {
            cur=edges[cur].last;
            to=edges[last].to;
            edges[last].to=i;
            edges[last].last=head2[to];
            head2[to]=last;
            last=cur;
        }
    }
    for(int i=1;i<=v;++i) head[i]=head2[i];
}
```
